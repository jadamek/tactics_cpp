#include <iostream>
#include <SFML/Graphics.hpp>
#include <vector>
#include <set>

class Obj
{
public:
    bool whatever;
    int id;
};

int main()
{
    sf::Clock timer;
    float elapsed;

    std::vector<Obj*> objs;
    for(int i = 0; i < 10000; i++)
    {
        objs.push_back(new Obj);
    }    
    timer.restart();            
    for(int i = 0; i < 10000; i++)
    {
        objs[i];
    }
    elapsed = timer.restart().asMicroseconds();

    std::cout << "Execution time: " << elapsed << " us " << std::endl;

    return 0;
}
